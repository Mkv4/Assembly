#<汇编代码：交换两个数>
_data 10,[5,6]
move R15,10000   #R15表示fp
move sp,R15    #使初始sp和fp指向同一位置

move R5,10     #存储首地址
load R2,00(R5)
load R3,01(R5)

push R3
push R2
call Lswap
goto Lprint

Lswap:
push R15       #将旧的fp值存入栈内
move R15,sp    # 新的fp等于sp
push R2
push R3
load R2,03(R15)
load R3,02(R15)
store 00(R5),R2
store 01(R5),R3

Lreturn:
pop R3        #返回初始的R3
pop R2        #返回初始的R2
move sp,R15   #sp = fp
pop R15       #重设fp的值，成为旧的fp
ret           #pop pc

Lprint:
_pr (10)
_pr (11)
